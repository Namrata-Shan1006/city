üö¶ City-wide Traffic Signal Optimization System

This system dynamically adjusts traffic light timings across urban intersections to minimize congestion and wait times. Real-time data from sensors helps balance vehicle flow for smoother city mobility.

üéØ SDG 11: Sustainable Cities and Communities

Algorithms Used

Primary Purpose

Shortest path coordination & network connectivity

Real-World Impact

Reduces congestion, improves urban mobility

C++ Implementation 
#include <stdio.h>

// Define graph size and constants
#define MAXN 4 // Using 4 vertices (V) from the original problem
#define INF 1000000 // A large number to represent infinity

// --- Global Data Structures ---
// cost[i][j] is the travel time/distance from i to j. INF if no direct path.
int cost[MAXN][MAXN] = {
    {0, 4, INF, INF},
    {4, 0, 8, INF},
    {INF, 8, 0, 7},
    {INF, INF, 7, 0}
};

int dist[MAXN]; // Stores the shortest distance from the source
int path[MAXN]; // Stores the predecessor of the node in the shortest path
int visited[MAXN]; // Acts as the sptSet (0 = false/not processed, 1 = true/processed)

// --- Dijkstra's Algorithm Implementation ---
void dijkstra(int n, int source) {
    // Initialization
    for(int i = 0; i < n; i++) {
        dist[i] = cost[source][i]; // Direct cost from source
        path[i] = source;          // Predecessor of i is source initially
        visited[i] = 0;
    }
    visited[source] = 1;
    dist[source] = 0; // Distance from source to itself is 0

    // Main loop: n-1 iterations to find the shortest path to all nodes
    for(int iter = 0; iter < n - 1; iter++) {
        int u = -1, min_dist = INF + 1;
        
        // 1. Find the unvisited node u with the smallest distance
        for(int i = 0; i < n; i++) {
            if(!visited[i] && dist[i] < min_dist) { 
                min_dist = dist[i]; 
                u = i; 
            }
        }

        if(u == -1) break; // All remaining nodes are unreachable or processed
        visited[u] = 1;

        // 2. Update distances to neighbors (Relaxation)
        for(int v = 0; v < n; v++) {
            // Check if v is not processed AND there is an edge from u to v
            if(!visited[v] && cost[u][v] != INF) {
                // Relaxation step: If the path through u is shorter than the current distance to v
                if(dist[u] + cost[u][v] < dist[v]) {
                    dist[v] = dist[u] + cost[u][v];
                    path[v] = u;
                }
            }
        }
    }
}

// --- Utility function to print the results ---
void print_results(int n, int source) {
    printf("Shortest Path Results from Source %d:\n", source);
    for (int i = 0; i < n; i++) {
        if (i == source) continue;
        
        printf("Vertex %d:\n", i);
        
        // Print Distance
        if (dist[i] == INF) {
            printf("  Distance: Unreachable\n");
        } else {
            printf("  Distance: %d\n", dist[i]);
            
            // Print Path (Reconstruct by tracing the path array backwards)
            printf("  Path: %d", i);
            int current = i;
            while (current != source) {
                current = path[current];
                printf(" <- %d", current);
            }
            printf("\n");
        }
    }
}

// --- Main Function ---
int main() {
    int num_vertices = MAXN;
    int source_vertex = 0;
    
    dijkstra(num_vertices, source_vertex);
    print_results(num_vertices, source_vertex);
    
    return 0;
}

#define V 5

int main() {
    int graph[V][V] = {{0, 6, INT_MAX, INT_MAX, INT_MAX},
                       {6, 0, 5, INT_MAX, 2},
                       {INT_MAX, 5, 0, 3, INT_MAX},
                       {INT_MAX, INT_MAX, 3, 0, 4},

C++ Implementation 
#include <iostream>
using namespace std;

void swap(int* a, int* b) {
    int t = *a; *a = *b; *b = t;
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = (low - 1);
    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return (i + 1);
}

void quickSort(int arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr)/sizeof(arr[0]);
    quickSort(arr, 0, n-1);
    cout << "Sorted inventory: ";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    cout << endl;
    return 0;
}

#include <iostream>
using namespace std;

int parent[10];

int find(int i) {
    if (parent[i] == -1) return i;
    return parent[i] = find(parent[i]);
}

void union_sets(int x, int y) {
    int xroot = find(x), yroot = find(y);
    if (xroot != yroot) parent[xroot] = yroot;
}

int main() {
    int n = 5;
    for (int i = 0; i < n; i++) parent[i] = -1;
    union_sets(0, 1); union_sets(1, 2); union_sets(3, 4);
    cout << "find(1): " << find(1) << endl;
    cout << "find(3): " << find(3) << endl;
    return 0;
}

#include <iostream>
using namespace std;

void merge(int arr[], int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    int L[n1], R[n2];
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];
    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(int arr[], int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        mergeSort(arr, l, m);
        mergeSort(arr, m + 1, r);
        merge(arr, l, m, r);
    }
}

int main() {
    int data[] = {38, 27, 43, 3, 9, 82, 10};
    mergeSort(data, 0, 6);
    cout << "Sorted flood data: ";
    for (int i = 0; i < 7; i++) cout << data[i] << " ";
    return 0;
}

#include <iostream>
using namespace std;

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++)
            if (arr[j] < arr[min_idx]) min_idx = j;
        swap(arr[min_idx], arr[i]);
    }
}

int main() {
    int skills[] = {85, 60, 90, 45, 75};
    selectionSort(skills, 5);
    cout << "Sorted skill scores: ";
    for (int i = 0; i < 5; i++) cout << skills[i] << " ";
    return 0;
}

#include <iostream>
using namespace std;

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (arr[j] > arr[j + 1]) swap(arr[j], arr[j + 1]);
}

int main() {
    int complaints[] = {3, 1, 4, 2, 5};
    bubbleSort(complaints, 5);
    cout << "Prioritized complaints: ";
    for (int i = 0; i < 5; i++) cout << complaints[i] << " ";
    return 0;
}
