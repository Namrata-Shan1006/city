Algorithms: K-d tree / spatial index for nearest neighbor, Dijkstra for routing
SDG 11: Sustainable Cities
Open example (JS snippet) 
// Find nearest vehicle by Haversine distance
function nearestVehicle(vehicles, incident){
  let best = null; let bestD = Infinity;
  for(v of vehicles){ d = haversine(v.lat,v.lon,incident.lat,incident.lon);
    if(d < bestD){ bestD = d; best = v; }}
  return best;
}
