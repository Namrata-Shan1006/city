#include <iostream> #include <queue> using namespace std; struct Route { int id, distance, safety; }; Route findSafeRoute(vector<Route> routes) { Route best = routes[0]; for (auto r : routes) { if (r.distance < best.distance && r.safety > best.safety) { best = r; } } return best; } int main() { vector<Route> available = {{1, 5, 8}, {2, 8, 9}, {3, 3, 6}}; Route evacRoute = findSafeRoute(available); cout << "Best Route ID: " << evacRoute.id << endl; return 0; }

Algorithm: Breadth-First Search (BFS) for Fault Isolation 1. Input: Grid node status (active/faulty) 2. Identify fault location using sensors 3. Perform BFS to map fault propagation 4. Isolate affected nodes using circuit breakers 5. Reroute power through alternate paths 6. Output: Recovery sequence and restored routes
üíª C++ Code
#include <iostream> #include <queue> #include <vector> using namespace std; void detectFault(vector<vector<int>>& grid) { queue<pair<int,int>> q; q.push({0, 0}); vector<vector<bool>> visited(grid.size(), vector<bool>(grid[0].size(), false)); while (!q.empty()) { auto [x, y] = q.front(); q.pop(); if (grid[x][y] == 0) continue; // faulty node visited[x][y] = true; } } int main() { vector<vector<int>> grid = {{1,1,0},{1,1,1},{0,1,1}}; detectFault(grid); return 0; }
