void push(int heap[], int *n, int val) {
  int i = ++(*n);
  heap[i] = val;
  while (i > 1 && heap[i/2] > heap[i]) {
    int t = heap[i]; heap[i] = heap[i/2]; heap[i/2] = t;
    i /= 2;
  }
}
