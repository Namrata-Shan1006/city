Smart Waste Collection — Dijkstra + Heap

Find shortest routes per vehicle (Dijkstra). Use a heap for scheduling tasks.

Open C++ snippet 
// anoosha_waste_dijkstra.cpp
#include <bits/stdc++.h>
using namespace std;
using pii = pair<int,int>;
vector<int> dijkstra(int n, vector>& adj, int src){
  const int INF=1e9; vector<int> dist(n,INF); dist[src]=0;
  priority_queue<pii, vector<pii>, greater<pii>> pq; pq.push({0,src});
  while(!pq.empty()){
    auto [d,u]=pq.top(); pq.pop();
    if(d!=dist[u]) continue;
    for(auto &e: adj[u]){ int v=e.first,w=e.second; if(dist[v]>d+w){ dist[v]=d+w; pq.push({dist[v],v}); } }
  }
  return dist;
}
