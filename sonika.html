University Admission — Union-Find (DSU)

Use DSU to manage allocation groups and constraints.

Open C++ snippet 
// sonika_admission_unionfind.cpp
#include <bits/stdc++.h>
using namespace std;
struct DSU{ vector p, r; DSU(int n):p(n),r(n,0){ iota(p.begin(), p.end(), 0); } int find(int a){ return p[a]==a?a:p[a]=find(p[a]); } void unite(int a,int b){ a=find(a); b=find(b); if(a==b) return; if(r[a]<r[b]) swap(a,b); p[b]=a; if(r[a]==r[b]) r[a]++; } };
int main(){ DSU d(10); d.unite(1,2); cout << d.find(2) << '\\n'; return 0; }
